test "Co-op Relay Smoke A"
	status partial
	description "Player A moves through normal gameplay while co-op emits relay presence."
	sequence
		inject "Three Earthly Barges Save"
		call "Load First Savegame"
		call "Co-op Relay Wait Connected"
		call "Co-op Relay Wait Peer"
		call "Co-op Relay Wait Direct Peer"
		call "Depart"
		assert
			"flagship landed" == 0
			"flagship disabled" == 0
			"flagship shields" == 600
			"flagship hull" == 1000
		call "Co-op Relay Wait Control Freeze Window"
		assert
			"flagship landed" == 0
			"flagship disabled" == 0
			"flagship shields" == 600
			"flagship hull" == 1000
		call "Co-op Relay Wait A Few Frames"
		navigate
			travel "Alpha Centauri"
		input
			command jump
		label "not alpha"
		branch "not alpha"
			not "flagship system: Alpha Centauri"
		assert
			"flagship fuel" < 300
		call "Co-op Relay Wait A Few Frames"
		navigate
			travel Sol
		input
			command jump
		label "not sol"
		branch "not sol"
			not "flagship system: Sol"
		navigate
			"travel destination" Earth
		call "Land"
		label "not refueled"
		branch "not refueled"
			"flagship fuel" < 300
		call "Co-op Relay Wait A Few Frames"


test "Co-op Relay Smoke B"
	status partial
	description "Player B observes Player A through relay presence without local gameplay mutation."
	sequence
		inject "Three Earthly Barges Save"
		call "Load First Savegame"
		call "Co-op Relay Wait Connected"
		call "Co-op Relay Wait Peer"
		call "Co-op Relay Wait Direct Peer"
		label "remote flying sol"
		branch "remote flying sol"
			"coop remote players flying in system: Sol" == 0
		assert
			"flagship planet: Earth" == 1
			"coop remote minimum fuel" == 1
		label "remote alpha"
		branch "remote alpha"
			"coop remote players in system: Alpha Centauri" == 0
		label "remote burned fuel"
		branch "remote burned fuel"
			"coop remote minimum fuel" >= 1
		assert
			"flagship planet: Earth" == 1
			"coop recent event: jumped" > 0
			"coop recent event: resources_changed" > 0
		label "remote returned landed"
		branch "remote returned landed"
			"coop remote players landed on: Earth" == 0
		label "remote refueled"
		branch "remote refueled"
			"coop remote minimum fuel" < 1
		assert
			"flagship planet: Earth" == 1
			"coop remote minimum fuel" == 1
			"coop recent event: landed" > 0
